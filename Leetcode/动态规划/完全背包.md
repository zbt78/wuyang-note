

### 二维数组

```c++
    vector<vector<int>> dp(V+1,vector<int>(N+1));
    
    for(int i = 1; i <= V; i++) {
        for(int j = 1; j <= N; j++) {
            if(i - w[j-1] < 0 ) {
                dp[i][j] = dp[i][j-1];
            } else {
                dp[i][j] = max(dp[i][j-1],dp[i-w[j-1]][j] + v[j-1]);   
                // // 和01背包有很小的区别
                // dp[i][j] = max(dp[i][j-1],dp[i-w[j-1]][j-1]+v[j-1]);
            }
        }
    }
    cout<<dp[V][N];
```



### 一维数组

```c++
    vector<int> dp(V+1);
    for(int i = 0; i < N; i++) {
        for(int j = 1; j <= V; j++) {
            if(j - w[i] < 0) continue;
            dp[j] = max(dp[j], dp[j-w[i]] + v[i]);
        }
    }
    cout<<dp[V];
```
